let allSpotData = {}; // 存储景点数据 // 页面加载时读取 JSON 数据 window.onload = function() { loadData(); }; async function loadData() { const response = await fetch('spots.json'); allSpotData = await response.json(); document.getElementById('loadingStatus').innerText = `? 数据加载完成，共 ${Object.keys(allSpotData).length} 个景点。`; createInputs(); // 默认生成3天 document.getElementById('mainContent').style.display = 'block'; } // 生成输入行 function createInputs() { const n = +document.getElementById("days").value; const area = document.getElementById("inputArea"); area.innerHTML = ""; for (let i = 1; i <= n; i++) { area.innerHTML += `
第${i}天：
`; } } // 模糊匹配 function fuzzyMatchSpot(inputName) { if (!inputName.trim()) return null; const input = inputName.trim().toLowerCase().replace(/\s+/g, ''); const spotNames = Object.keys(allSpotData); let bestMatch = null; let bestScore = 0; for (const name of spotNames) { const spotName = name.toLowerCase().replace(/\s+/g, ''); let score = similarity(input, spotName); if (score > bestScore) { bestScore = score; bestMatch = name; } } if (bestScore > 0.3) { // 设定匹配阈值 return { name: bestMatch, ...allSpotData[bestMatch] }; } return null; } // 简单相似度计算 function similarity(a, b) { a = a.toLowerCase(); b = b.toLowerCase(); let longer = a.length > b.length ? a : b; let shorter = a.length > b.length ? b : a; let same = 0; for (let i = 0; i < shorter.length; i++) { if (longer.includes(shorter[i])) same++; } return same / longer.length; } // 生成行程 function build() { const lines = document.querySelectorAll(".day-line"); const out = document.getElementById("output"); let html = ""; lines.forEach((line, idx) => { const p1 = line.querySelector(".p1").value.trim(); const p2 = line.querySelector(".p2").value.trim(); const p3 = line.querySelector(".p3").value.trim(); const inputs = [p1, p2, p3].filter(Boolean); if (inputs.length === 0) return; html += `
第${idx + 1}天
`; inputs.forEach(input => { const matchedSpot = fuzzyMatchSpot(input); if (matchedSpot) { html += `
${matchedSpot.name}
? 已匹配到景点：${matchedSpot.name}
${matchedSpot.desc}
`; } else { html += `
${input}
? 未匹配到对应景点，请检查名称
暂无该景点的描述
`; } }); html += `
`; }); out.innerHTML = html; window.finalHTML = html; // 保存生成内容用于导出 } // 导出 Word 表格 function downloadWord() { if (!window.finalHTML) { alert("请先生成行程"); return; } const wordContent = `

旅游行程单

${window.finalHTML.replace(/
') .replace(/<\/div>\s*$/, '')} `; const blob = new Blob([wordContent], { type: "application/msword" }); const a = document.createElement("a"); a.href = URL.createObjectURL(blob); a.download = "旅游行程表.doc"; a.click(); }
